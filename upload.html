async function handleUpload() { const vInput = document.getElementById('video-input'); const videoFile = vInput.files[0]; if(!videoFile) return sojiAlert("দয়া করে একটি ভিডিও সিলেক্ট করুন!"); // ইউজার চেক const user = auth.currentUser; if(!user) return sojiAlert("আপলোড করতে আগে লগইন করুন!"); const btn = document.getElementById('btn-post'); const progressBox = document.getElementById('progress-box'); const progressFill = document.getElementById('progress-fill'); const progressPercent = document.getElementById('p-percent'); btn.disabled = true; btn.innerText = "প্রসেসিং..."; progressBox.style.display = 'block'; // ১. ক্লাউডিনারি ডাটা প্রিপারেশন const formData = new FormData(); formData.append('file', videoFile); formData.append('upload_preset', 'sojida_videos'); // আপনার দেওয়া প্রেসেট formData.append('cloud_name', 'dlpwdx288'); // আপনার দেওয়া ক্লাউড নাম // ২. আপলোড ইঞ্জিন (XHR) const xhr = new XMLHttpRequest(); xhr.open('POST', 'https://api.cloudinary.com/v1_1/dlpwdx288/video/upload', true); xhr.upload.onprogress = (e) => { if (e.lengthComputable) { const percent = Math.round((e.loaded / e.total) * 100); progressFill.style.width = percent + '%'; progressPercent.innerText = percent + '%'; btn.innerText = "আপলোড হচ্ছে... " + percent + "%"; } }; xhr.onload = async () => { if (xhr.status >= 200 && xhr.status < 300) { const response = JSON.parse(xhr.responseText); // ৩. ফায়ারবেস স্টোর (মগজে ডাটা সেভ) try { await db.collection("videos").add({ url: response.secure_url, caption: document.getElementById('caption').value || "", category: document.getElementById('category').value, creatorId: user.uid, creatorName: user.displayName || "SojiDa Creator", likes: 0, views: 0, likedBy: [], createdAt: firebase.firestore.FieldValue.serverTimestamp() }); sojiAlert("ভিডিও পাবলিশ হয়েছে!", "সফল!"); setTimeout(() => location.href = 'index.html', 2000); } catch (dbError) { console.error("Database Error:", dbError); sojiAlert("ফায়ারবেসে ডাটা সেভ হয়নি!"); btn.disabled = false; } } else { console.error("Cloudinary Error:", xhr.responseText); sojiAlert("ক্লাউডিনারি আপলোড রিজেক্ট করেছে! প্রেসেট 'Unsigned' কি না চেক করুন।"); btn.disabled = false; btn.innerText = "আবার চেষ্টা করুন"; } }; xhr.onerror = () => { sojiAlert("ইন্টারনেট কানেকশন চেক করুন!"); btn.disabled = false; }; xhr.send(formData); }
